	public static boolean arePermutations(int[] array1, int[] array2)
	{
		int length = array1.length;
		int check;

		for(int i = 0; i < array1.length; i++)
		{
			for(int j = 0; j < array2.length; j++)
			{
				if(array1[i] == array2[j])
				{
					check++;
					break;
				}
			}
		}

		if(check == length)
		{
			return true;
		}else
		{
			return false;
		}
	}
